vue.config.js

发表于 2020-03-03 更新于 2019-11-29

vue-cli3.0版本搭建项目时需手动配置vue.config.js文件，要重点说明跨域问题，以及前端的详细配置

前端向后端请求数据需要proxy那一段的配置，也可在本地使用mock模拟随机数据，在本地测试，详情需要仔细反复看vue-cli里面关于代理的配置很详细，但是可能我脑子不好使，要看好久才理解。。。。。

在项目的根目录新建vue.config.js文件，配置如下
'use strict'
const path = require('path')

function resolve(dir) {
    return path.join(__dirname, dir)
}

const port = process.env.port || process.env.npm_config_port || 8888       //dev port


module.exports = {
    // 没有具体地址，使用'/'能正确处理嵌套路由问题
    publicPath: '/',
    // 运行npm run build 时生成的生产环境构建文件的目录
    outputDir: 'dist',
    // 在开发环境下通过eslint-loader 在每次保存时lint代码
    lintOnSave: process.env.NODE_ENV === 'development',
    // 生产环境的 source map
    productionSourceMap: false,
    // 热部署，配置同webpack-dev-server
    devServer: {
        // 端口号
        port: port,
        // 开发自动打开
        open: true,
        // 没的说
        openPage: '',
        // 当出现编译器错误或警告时
        // 在浏览器中显示全屏覆盖层
        overlay: {
            warnings: false,
            errors: true
        },
        // 开发代理
        // 这里被折磨一周，前后端分离，请求数据出现跨域，脑子有问题，琢磨了好多天
        proxy: {
            '/agilefast-admin': {
                target: 'http://10.1.102.120:8080',
                pathRewrite: {
                    '^/agilefast-admin': '/agilefast-admin'
                },
                // target是域名的话，需要这个参数
                changeOrigin: true,
                //  设置支持https协议的代理
                secure: false
            }
        }
        // before: require('./mock/server')
    },
    configureWebpack: {
        // 这些玩意使用CDN
        externals: {
            'vue': 'Vue',
              'vuex': 'Vuex',
              'vue-router': 'VueRouter',
              'nprogress': 'NProgress',
              'animate ': 'animate.css',
              'element-ui': 'ELEMENT'
        },
        resolve:{
            alias: {
                '@': resolve('src')
            }
        }
    }
}
