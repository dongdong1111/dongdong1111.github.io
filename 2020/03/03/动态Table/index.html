动态Table

应项目需求，做出一个类似数据库写入SQL语句执行，能够查出对应表中的所有数据，需要在前端页面创建一个动态表格，因为啥要动态的呢？？是这样似的，每一张表中的字段名称不一样，所以呢table表格中的表头也不能一样。。。那么效果如何实现呢？？？？我是这样式写的。。。。。

首先在页面中需要写一个容器装这个table表格，代码如下
<div id="tableContainer">

</div>
其次要对它进行操作那么在js文件需要完善，代码如下
//动态创建table数据表格
function createTable(rows,cols) {
    this.rows = rows;
    this.cols = cols;
    // 获取页面中存放table表格的容器
    // 创建一个table
    // 创建table的表头thead
    // 创建table的表中内容 tbody
    var tableContainer = document.getElementById('tableContainer');
    var Table = document.createElement('table');
    var Thead = document.createElement('thead');
    var Tbody = document.createElement('tbody);
    // 此处可设置一些table的样式
    Table.setAttribute('border','1');
    Table.setAttribute('width','100%');

    //此处根据循环确定生成表格的行(thead)
    //此处做一下说明：为什么i<1?? 此处考虑到表头只需要一行，所以不需要一直循环
    //第二处循环key值是因为想确定有多少列，此项目数据使用本地json文件，那么需要借用key值来确定有多少列，，，为什么罗里吧嗦，是怕自己以后看不懂，，，哈哈哈哈哈
    for(var i = 0; i < 1; i++){
        var trHead = document.createElement('tr');
        Thead.appendChild(trHead);
        // 循环的是列
        // 创建单元格
        // 把属性值赋值给单元格td
        for(var key in jsonArray[0]){
            var th = document.createElement('td');
            trHead.appendChild(th);
            th.innerHTML = key;
        }
    }
    // 此处循环生成表格的行(tbody)
    // 此处i<this.rows是因为tbody里有多少行不确定，要看返回的json数据里有多少条数据，那么有多少条就代表有多少行
    // 第二处循环是确定每一列里的value值
    for(var i = 0; i < this.rows; i++){
        var tr = document.createElement('tr');
        Tbody.appendChild(tr);
        for(var k in jsonArray[i]){
            var td = document.createElement('td');
            tr.appendChild(td);
            td.innerHTML = jsonArray[i][k];
        }
    }
    Table.append(Thead,Tbody);
    // 生成table表格
    tableContainer.appendChild(Table);
}
最后点击某一处的时候出现table表格，那么就绑定一个点击事件，调用上边的方法
createTable(1000,1000);
按正常逻辑来说这个地方不可以写死，，
以后继续完善吧，目前先这样式的吧！

那么这样操作会遇到什么问题呢？？那就是每次点击都出现一个table表格，这肯定是不对滴，那我是这样式想的

当我点击按钮的时候我就先判断一下子这个页面中的容器有没有table表格，如果没有那我给你加一个，如果你有，那么就把你清空，重新给你一个
我是这样式写的，，对不对不确定，但是是我要的效果，因为数据都是本地的，暂时也没发现啥问题...代码如下
